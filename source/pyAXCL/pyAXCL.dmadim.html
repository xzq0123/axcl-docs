12.8.2. types#

AX_DMADIM_ENDIAN_E#

dmadim endian

AX_DMADIM_ENDIAN_DEF = 0
AX_DMADIM_ENDIAN_32 = 1
AX_DMADIM_ENDIAN_16 = 2
AX_DMADIM_ENDIAN_MAX = 2
AX_DMADIM_XFER_MODE_E#

dmadim xfer mode

AX_DMADIM_1D = 0
AX_DMADIM_2D = 1
AX_DMADIM_3D = 2
AX_DMADIM_4D = 3
AX_DMADIM_MEMORY_INIT = 4
AX_DMADIM_CHECKSUM = 5
class AX_DMADIM_XFER_STAT_T#
dict_dmadim_xfer_stat = {
    "id": int,
    "checksum": int,
    "stat": int
}
class AX_DMADIM_INFO_T#
dict_dmadim_info = {
    "phy_addr": int,
    "img_w": int,
    "stride": int
}
class AX_DMADIM_DESC_XD_T#
dict_dmadim_desc_xd = {
    "n_tiles": list[int, int, int],
    "src_info": AX_DMADIM_INFO_T,
    "dst_info": AX_DMADIM_INFO_T,
}
class AX_DMADIM_DESC_T#
dict_dmadim_desc = {
    "phy_src": int,
    "phy_dst": int,
    "size": int
}
class AX_DMADIM_MSG_T#
dict_dmadim_msg = {
    "desc_buf": list of dict_dmadim_desc, AX_DMADIM_DESC_T,
    "endian": int,
    "desc_buf": int,
    "callback": callback funtion: def callback(stat, cb_arg)->None,  stat-AX_DMADIM_XFER_STAT_T,
    "cb_arg": int,
    "dma_mode": int
}
